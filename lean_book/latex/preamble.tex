% ---------------------------------------------------------------------
% Chapter/section styling
% ---------------------------------------------------------------------
\usepackage{titlesec}
\definecolor{ruleblue}{RGB}{40,70,120}
% No chapter-number label here: every chapter's title text already
% embeds its own "Chapter N:" (the book's own numbering, e.g. Chapter 0
% for 00-setup/, Chapter 6 for 06-groups/), which does not line up with
% LaTeX's automatic \thechapter counter (that counter increments once per
% \chapter{} call including the 00-setup chapter and the unnumbered
% Appendix, so it runs one or more ahead of the book's own numbers) --
% showing both would be redundant at best, contradictory at worst.
\titleformat{\chapter}[display]
  {\normalfont\bfseries}
  {}
  {0pt}
  {{\color{ruleblue}\titlerule[1pt]}\vspace{10pt}\Huge}
  [{\vspace{4pt}\color{ruleblue}\titlerule[0.5pt]}]
\titleformat{\section}
  {\normalfont\Large\bfseries\color{ruleblue}}{\thesection}{0.8em}{}
\titleformat{\subsection}
  {\normalfont\large\bfseries}{\thesubsection}{0.8em}{}
\titlespacing*{\chapter}{0pt}{0pt}{30pt}

% book class's default \cleardoublepage (used before every \chapter, since
% book defaults to openright) pads to an odd page with a bare \hbox{}+\newpage
% but does NOT set \thispagestyle{empty} on that inserted page, so it still
% shows the fancyhdr header rule and page-number footer despite having no
% content -- fix it to render genuinely blank.
\makeatletter
\def\cleardoublepage{\clearpage\if@twoside \ifodd\c@page\else
  \hbox{}\thispagestyle{empty}\newpage\if@twocolumn\hbox{}\newpage\fi\fi\fi}
\makeatother
